224: /* Inline-hook a target function ONCE (guarded by *orig, which becomes the
225:  * trampoline that calls the original). INLINE hooking rewrites the function BODY,
226:  * so a third-party overlay (UE4SS) that re-patches the same vtable slot afterwards
227:  * cannot bypass us — the reason plain vtable hooking failed to capture TXR. */
228: static void arm_inline_once(void *fn, void *detour, void **orig, const char *name) {
229:     if (*orig) return;                    /* already inline-hooked */
230:     void *tramp = nullptr;
231:     if (rgpu_inline_hook(fn, detour, &tramp)) { *orig = tramp; rgpu_log("inline-hooked %s @ %p", name, fn); }
232:     else rgpu_log("inline-hook ABORTED for %s @ %p (prologue not relocatable; left unhooked)", name, fn);
233: }

235: /* Suspend every OTHER thread in the process so writing D3D12Core code pages can't
236:  * race a thread executing the function being patched (the crash that boot-looped
237:  * TXR). MinHook-style. While frozen we must NOT touch the CRT heap (malloc / fopen)
238:  * or we could deadlock on a lock a frozen thread holds — so no logging/allocation
239:  * there; rgpu_inline_hook uses only VirtualAlloc/Protect syscalls. */
240: static HANDLE g_frozen[4096]; static int g_nfrozen = 0;
241: static void rgpu_freeze_other_threads() {
242:     g_nfrozen = 0;
243:     DWORD pid = GetCurrentProcessId(), tid = GetCurrentThreadId();
244:     HANDLE snap = CreateToolhelp32Snapshot(TH32CS_SNAPTHREAD, 0);
245:     if (snap == INVALID_HANDLE_VALUE) return;
246:     THREADENTRY32 te; te.dwSize = sizeof(te);
247:     if (Thread32First(snap, &te)) do {
248:         if (te.th32OwnerProcessID == pid && te.th32ThreadID != tid) {
249:             HANDLE h = OpenThread(THREAD_SUSPEND_RESUME, FALSE, te.th32ThreadID);
250:             if (h) { if (SuspendThread(h) != (DWORD)-1 && g_nfrozen < 4096) g_frozen[g_nfrozen++] = h; else CloseHandle(h); }
251:         }
252:     } while (Thread32Next(snap, &te));
253:     CloseHandle(snap);
254: }
255: static void rgpu_resume_other_threads() {
256:     for (int i = 0; i < g_nfrozen; i++) { ResumeThread(g_frozen[i]); CloseHandle(g_frozen[i]); }
257:     g_nfrozen = 0;
258: }
259: struct HookItem { void *fn; void *detour; void **orig; const char *name; };
260: static void install_batch_frozen(HookItem *items, int n) {
261:     if (!n) return;
262:     rgpu_freeze_other_threads();
263:     for (int i = 0; i < n; i++) { void *tr = nullptr; if (rgpu_inline_hook(items[i].fn, items[i].detour, &tr)) *items[i].orig = tr; }
264:     rgpu_resume_other_threads();
265:     for (int i = 0; i < n; i++) rgpu_log("%s %s @ %p", *items[i].orig ? "inline-hooked" : "inline-hook ABORTED for", items[i].name, items[i].fn);
266: }
