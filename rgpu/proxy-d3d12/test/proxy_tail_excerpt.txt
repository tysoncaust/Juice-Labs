313: /* Inline-hook device methods from the real returned interface. This survives
314:  * UE4SS/overlay vtable replacement just like the queue/list inline hooks. */
315: static void arm_from_device(ID3D12Device *dev, const char *which) {
316:     if (!dev) return;
317: 
318:     ID3D12Device *base = nullptr;
319:     ID3D12Device4 *d4 = nullptr;
320:     ID3D12Device9 *d9 = nullptr;
321:     dev->QueryInterface(__uuidof(ID3D12Device), (void **)&base);
322:     dev->QueryInterface(__uuidof(ID3D12Device4), (void **)&d4);
323:     dev->QueryInterface(__uuidof(ID3D12Device9), (void **)&d9);
324: 
325:     void *bv = base ? *(void **)base : nullptr;
326:     void *v4 = d4 ? *(void **)d4 : nullptr;
327:     void *v9 = d9 ? *(void **)d9 : nullptr;
328: 
329:     HookItem items[5]; int n = 0;
330:     auto add = [&](void *obj, unsigned slot, void *detour, void **orig, const char *name) {
331:         if (obj && !*orig && n < 5) items[n++] = { vslot(obj, slot), detour, orig, name };
332:     };
333:     add(base, RGPU_SLOT_Device_CheckFeatureSupport, (void *)h_CFS, (void **)&o_CFS,
334:         "Device::CheckFeatureSupport");
335:     add(base, RGPU_SLOT_Device_CreateCommandQueue, (void *)h_DevCCQ, (void **)&o_DevCCQ,
336:         "Device::CreateCommandQueue");
337:     add(base, RGPU_SLOT_Device_CreateCommandList, (void *)h_DevCCL, (void **)&o_DevCCL,
338:         "Device::CreateCommandList");
339:     add(d4, RGPU_SLOT_Device4_CreateCommandList1, (void *)h_DevCCL1, (void **)&o_DevCCL1,
340:         "Device4::CreateCommandList1");
341:     add(d9, RGPU_SLOT_Device9_CreateCommandQueue1, (void *)h_DevCCQ1, (void **)&o_DevCCQ1,
342:         "Device9::CreateCommandQueue1");
343:     install_batch_frozen(items, n);
344: 
345:     /* The conservative inline decoder can reject some base-method prologues.
346:      * Keep a safe vtable fallback for those exact methods; the Agility 1 methods
347:      * remain inline-hooked and therefore survive later overlay vtable changes. */
348:     bool fbCFS = false, fbCCQ = false, fbCCL = false, fbCCQ1 = false, fbCCL1 = false;
349:     if (base && !o_CFS) fbCFS = patch_slot(base, RGPU_SLOT_Device_CheckFeatureSupport,
350:                                             (void *)h_CFS, (void **)&o_CFS);
351:     if (base && !o_DevCCQ) fbCCQ = patch_slot(base, RGPU_SLOT_Device_CreateCommandQueue,
352:                                               (void *)h_DevCCQ, (void **)&o_DevCCQ);
353:     if (base && !o_DevCCL) fbCCL = patch_slot(base, RGPU_SLOT_Device_CreateCommandList,
354:                                               (void *)h_DevCCL, (void **)&o_DevCCL);
355:     if (d9 && !o_DevCCQ1) fbCCQ1 = patch_slot(d9, RGPU_SLOT_Device9_CreateCommandQueue1,
356:                                                (void *)h_DevCCQ1, (void **)&o_DevCCQ1);
357:     if (d4 && !o_DevCCL1) fbCCL1 = patch_slot(d4, RGPU_SLOT_Device4_CreateCommandList1,
358:                                                (void *)h_DevCCL1, (void **)&o_DevCCL1);
359: 
360:     rgpu_log("armed-hybrid %s device input=%p base=%p/vtbl=%p d4=%p/vtbl=%p d9=%p/vtbl=%p originals CFS=%p CCQ=%p CCL=%p CCQ1=%p CCL1=%p fallback=%d/%d/%d/%d/%d",
361:              which, (void *)dev, (void *)base, bv, (void *)d4, v4, (void *)d9, v9,
362:              (void *)o_CFS, (void *)o_DevCCQ, (void *)o_DevCCL,
363:              (void *)o_DevCCQ1, (void *)o_DevCCL1,
364:              fbCFS, fbCCQ, fbCCL, fbCCQ1, fbCCL1);
365: 
366:     if (d9) d9->Release();
367:     if (d4) d4->Release();
368:     if (base) base->Release();
369: }

371: static bool adapter_is_warp(IUnknown *adapter) {
372:     if (!adapter) return false;   /* null == default hardware adapter */
373:     IDXGIAdapter *a = nullptr;
374:     if (SUCCEEDED(adapter->QueryInterface(__uuidof(IDXGIAdapter), (void **)&a)) && a) {
375:         DXGI_ADAPTER_DESC d{}; a->GetDesc(&d); a->Release();
376:         return d.VendorId == 0x1414;   /* Microsoft Basic Render Driver (WARP) */
377:     }
378:     return false;
379: }

394: static bool hook_factory_object(void *obj) {
395:     if (!obj) return false;
396:     ID3D12DeviceFactory *factory = nullptr;
397:     if (FAILED(((IUnknown *)obj)->QueryInterface(__uuidof(ID3D12DeviceFactory), (void **)&factory)) || !factory)
398:         return false;
399:     void *vtbl = *(void **)factory;
400:     bool ok = true;
401:     if (vtbl != g_hooked_factory_vtbl) {
402:         ok = patch_slot(factory, RGPU_SLOT_Factory_CreateDevice,
403:                         (void *)h_FactoryCreateDevice, (void **)&o_FactoryCreateDevice);
404:         if (ok) {
405:             g_hooked_factory_vtbl = vtbl;
406:             rgpu_log("hooked ID3D12DeviceFactory::CreateDevice factory=%p vtbl=%p",
407:                      (void *)factory, vtbl);
408:         }
409:     }
410:     factory->Release();
411:     return ok;
412: }
